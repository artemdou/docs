## Erforderliche Ressourcen

Für die einfache Implementierung sind keine zusätzlichen Ressourcen erforderlich.

## Beispielcode

Fügen Sie den folgenden Code in das HTML Ihrer Website ein.

```markdown iframe.html
<iframe src="https://demo.nets-pay.link/pay?tid=345a142d" width="530" height="700">
</iframe>
```

Für __src__ müssen Sie die URL Ihrer Zahlungsseite eingeben.

## Erweiterte Integration

```javascript Code Snippet
var iFrame = jQuery('iframe');

var postMessage = function(e) {
  if (typeof e.data === 'string') {
    try {
      var data = JSON.parse(e.data);
    } catch (e) {}
    if (data && data.nets) {
      jQuery.each(data.nets, function(name, value) {
        switch (name) {
          // implement functions here
        }
      });
    }
  }
};

iFrame.on('load', function() {
        jQuery(this)\[0\].contentWindow.postMessage(JSON.stringify({origin: window.location.origin}), iFrame.attr('src'));
});

window.addEventListener('message', postMessage, false);
```

Sie sollten die folgenden Funktionen implementieren (an der Stelle des Kommentars):

- __height__: Dynamische Höhe des iframes. Hilft zu vermeiden, dass der iframe einen eigenen Scrollbalken hat.
- Sie sollten den Parameter `appview ` nicht setzen, wenn Sie eine dynamische Höhe wünschen.
-
- __top__: Wenn Sie die dynamische __height__-Funktionalität implementiert haben, müssen Sie wahrscheinlich auch die Scroll-Funktionalität implementieren (z. B. bei der Schaltfläche „Add to cart" oder „Go to checkout").
-
- __transaction__: Ermöglicht es Ihnen, das Transaktionsobjekt mit den Transaktionsdetails wie Status zu erhalten.
-
- __closeModal__: Dies enthält die Redirect-URL, die definiert, zu welcher Seite Ihr Kunde geführt wird. Der iFrame darf keine Weiterleitungen auf Ihrer Seite vornehmen, daher müssen Sie diese Weiterleitung selbst ausführen.

Hier ist ein vollständiges Codebeispiel:

```javascript Complete Code Example
var iFrame = jQuery('iframe');
var transaction = null;
var lastPostMessageHeight = 0;

var updateIframeHeight = function() {
  var height = lastPostMessageHeight;

  // This can be individually changed.
  // Currently the full height is only applied for big displays (> 590px in width)
  if (window.innerWidth <= 590) {
    iFrame.css('height', '100%');
  } else if (height) {
    iFrame.css('height', height + 'px');
  }
};

var scrollPage = function(offset) {
  var positionToScrollTo = iFrame.position().top + offset;
  jQuery('body, html').animate({scrollTop: positionToScrollTo}, 1000, 'linear', function() {
    jQuery(window).trigger('scroll')
  });
};

var postMessage = function(e) {
  if (typeof e.data === 'string') {
    try {
      var data = JSON.parse(e.data);
    } catch (e) {}
    if (data && data.nets) {
      jQuery.each(data.nets, function(name, value) {
        switch (name) {
          case 'height':
            lastPostMessageHeight = parseInt(value);
            updateIframeHeight();
            break;
          case 'top':
            scrollPage(parseInt(value));
            break;
          case 'transaction':
            if (typeof value === 'object') {
              transaction = value;
            }
            break;
          case 'closeModal':
            window.location.href = value;
            break;
        }
      });
    }
  }
};

window.addEventListener('message', postMessage, false);

iFrame.on('load', function() {
  jQuery(this)[0].contentWindow.postMessage(JSON.stringify({origin: window.location.origin}), iFrame.attr('src'));
  jQuery(window).resize(updateIframeHeight);
  updateIframeHeight();
});

var t;
jQuery(window).scroll(function(event) {
  window.clearTimeout(t);
  t = window.setTimeout(function() {
    iFrame[0].contentWindow.postMessage(JSON.stringify({scrollTopShoppingCart: jQuery(window).scrollTop() - iFrame.position().top}), iFrame.attr('src'));
  }, 100);
});
```
